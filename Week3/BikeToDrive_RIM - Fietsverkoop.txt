CREATE TABLE Fiets_Verkoop_Fabrikant(
    fabrikantnr INT,
    naam VARCHAR(45),
    adres VARCHAR(100),
    plaats VARCHAR(45),

    PRIMARY KEY(fabrikantnr)
);

CREATE TABLE Fiets_Verkoop_Fiets(
    fietsnr INT,
    soort VARCHAR(45),
    merk VARCHAR(45),
    type VARCHAR(45),
    standaardprijs FLOAT,
    inkoopprijs FLOAT,
    kleur VARCHAR(45),
    fabrikant INT NOT NULL,

    PRIMARY KEY(fietsnr),
    FOREIGN KEY(fabrikant) REFERENCES Fiets_Verkoop_Fabrikant(fabrikantnr)
);

CREATE TABLE Fiets_Verkoop(
    fiets_verkoopnr INT,
    datum DATE,
    aantal INT,
    verkoopprijs FLOAT,
    klant INT NOT NULL, 
    fiets INT NOT NULL, 
    monteur INT NOT NULL,

    PRIMARY KEY(fiets_verkoopnr),
    FOREIGN KEY(klant) REFERENCES Klant(klantnr),
    FOREIGN KEY(fiets) REFERENCES Fiets(fietsnr),
    FOREIGN KEY(monteur) REFERENCES Monteur(monteurnr)
);

CREATE TABLE Fiets_Verkoop_Klant(
    klantnr INT,
    naam VARCHAR(45),
    adres VARCHAR(45),
    woonplaats VARCHAR(45),
    geslacht VARCHAR(45),
    geboortedatum DATE,

    PRIMARY KEY(klantnr)
);

CREATE TABLE Fiets_Verkoop_Monteur(
    monteurnr INT,
    naam VARCHAR(45),
    woonplaats VARCHAR(45),
    uurloon FLOAT,
    filiaal INT NOT NULL,

    PRIMARY KEY(monteurnr),
    FOREIGN KEY(filiaal) REFERENCES Filiaal(filiaalnr)
);

CREATE TABLE Fiets_Verkoop_Filiaal(
    filiaalnr INT,
    naam VARCHAR(45),
    adres VARCHAR(45),
    postcode VARCHAR(45),

    PRIMARY KEY(filiaalnr)
);
